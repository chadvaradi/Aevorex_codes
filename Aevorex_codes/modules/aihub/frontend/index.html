Quick Start

import aevorex

# Initialize the client
client = aevorex.Client(api_key="your_api_key")

# Make a prediction
response = client.models.predict(
    model="gpt-4-turbo",
    prompt="Analyze this text for sentiment",
    max_tokens=100
)

print(response.result)
